# Admin UI 样式系统规范

> 基于鸿蒙项目 UI 规范，适配 Ant Design + React 技术栈

---

## 一、颜色系统

### 1.1 主题色配置（ConfigProvider）

```tsx
// src/theme/index.ts
import type { ThemeConfig } from 'antd';

export const theme: ThemeConfig = {
  token: {
    // 主色
    colorPrimary: '#3277FF',
    colorPrimaryHover: '#5A94FF',
    colorPrimaryActive: '#1A5FE6',

    // 成功色
    colorSuccess: '#52C41A',

    // 警告色
    colorWarning: '#FAAD14',

    // 错误色
    colorError: '#ED184E',
    colorErrorHover: '#FF3D6E',

    // 文字色
    colorText: '#333333',
    colorTextSecondary: '#84888F',
    colorTextTertiary: '#999999',
    colorTextQuaternary: '#CCCCCC',

    // 边框色
    colorBorder: '#E5E5E5',
    colorBorderSecondary: '#F0F0F0',

    // 背景色
    colorBgContainer: '#FFFFFF',
    colorBgLayout: '#F1F2F7',
    colorBgElevated: '#FFFFFF',

    // 圆角
    borderRadius: 4,
    borderRadiusLG: 8,
    borderRadiusSM: 2,

    // 字体
    fontSize: 14,
    fontSizeLG: 16,
    fontSizeSM: 12,
  },
};
```

### 1.2 语义化颜色变量

```less
// src/styles/variables.less

// === 品牌色 ===
@color-primary: #3277FF;
@color-primary-light: #E8F0FF;
@color-primary-dark: #1A5FE6;

// === 功能色 ===
@color-success: #52C41A;
@color-warning: #FAAD14;
@color-error: #ED184E;
@color-info: #3277FF;

// === 文字色（与 H5 统一） ===
@color-text-primary: #333333;
@color-text-secondary: #84888F;
@color-text-tertiary: #999999;
@color-text-disabled: #CCCCCC;
@color-text-white: #FFFFFF;

// === 背景色（与 H5 统一） ===
@color-bg-page: #F1F2F7;
@color-bg-container: #FFFFFF;
@color-bg-hover: #F5F5F5;
@color-bg-disabled: #F5F5F5;

// === 边框色 ===
@color-border: #E5E5E5;
@color-border-light: #F0F0F0;

// === 状态色 ===
@color-status-pending: #FAAD14;
@color-status-success: #52C41A;
@color-status-failed: #ED184E;
@color-status-disabled: #999999;
```

### 1.3 H5 到 Admin 颜色映射

| H5 变量 | Admin 变量 | 色值 | 用途 |
|---------|-----------|------|------|
| `$text-blue` | `@color-primary` | #3277FF | 主色、链接 |
| `$text-grey` | `@color-text-secondary` | #84888F | 次要文字 |
| `$text-red` | `@color-error` | #ED184E | 错误、必填标记 |
| `$bgc-grey` | `@color-bg-page` | #F1F2F7 | 页面背景 |
| `$bgc-blue` | `@color-primary` | #3377FF | 主按钮背景 |

## 二、间距系统

### 2.1 基础间距（rpx → px 转换）

| H5 (rpx) | Admin (px) | 变量名 | 用途 |
|----------|------------|--------|------|
| 8rpx | 4px | `@spacing-xs` | 紧凑间距 |
| 16rpx | 8px | `@spacing-sm` | 小间距 |
| 20rpx | 10px | `@spacing-md` | 常用间距 |
| 24rpx | 12px | `@spacing-lg` | 大间距 |
| 32rpx | 16px | `@spacing-xl` | 页面边距 |
| 40rpx | 20px | `@spacing-xxl` | 区块间距 |

```less
// src/styles/variables.less

@spacing-xs: 4px;
@spacing-sm: 8px;
@spacing-md: 10px;
@spacing-lg: 12px;
@spacing-xl: 16px;
@spacing-xxl: 20px;
```

### 2.2 布局间距

```less
// 页面级
@page-padding: 24px;
@section-gap: 24px;

// 卡片级
@card-padding: 16px;
@card-gap: 16px;

// 表单级
@form-item-gap: 24px;
@form-label-gap: 8px;
```

## 八、使用示例

### 8.1 ConfigProvider 配置

```tsx
// src/App.tsx
import { ConfigProvider } from 'antd';
import { theme } from './theme';
import zhCN from 'antd/locale/zh_CN';

function App() {
  return (
    <ConfigProvider theme={theme} locale={zhCN}>
      <RouterProvider router={router} />
    </ConfigProvider>
  );
}
```

### 8.2 组件样式使用

```less
// src/pages/tenant/index.module.less
@import '@/styles/variables.less';

.container {
  padding: @page-padding;
  background: @color-bg-page;
}

.card {
  padding: @card-padding;
  background: @color-bg-container;
  border-radius: @border-radius-base;
  box-shadow: @shadow-card;

  &:hover {
    box-shadow: @shadow-hover;
  }
}

.title {
  font-size: @font-size-lg;
  font-weight: @font-weight-medium;
  color: @color-text-primary;
}

.description {
  font-size: @font-size-base;
  color: @color-text-secondary;
  margin-top: @spacing-sm;
}
```
